# 2a-CaTT-V9: Core CaTT over preCaTT

This module gives the intrinsic CaTT syntax built directly over the raw
preCaTT layer.

The key organizational point is that the file has two layers.

1. A top-level raw-indexed layer works over raw contexts `Pre.Ctx`.
   There we define the well-formedness predicates
   `Ctx-iswf`, `Ty-iswf`, `Tm-iswf`, and `Sub-iswf`, together with the
   primed proof-carrying syntax `Ty'`, `Tm'`, and `Sub'`.
2. The outer part of the file then packages a raw context together with a
   well-formedness proof as the public type `Ctx`, and rebuilds the public
   `Ty`, `Tm`, and `Sub` over that packaged context.

This explains why many names appear twice when reading the code: first in the
raw-indexed proof layer, and then again in the public API over packaged
contexts. The point of V9 is that the raw-indexed layer is not hidden away in a
local module: later developments may refer to it directly whenever that gives a
cleaner proof.

Compared to `1a-preCaTT`, the CaTT syntax stores explicit typing/fullness
certificates:
- endpoint typing proofs for the two endpoints of a hom-type
- typing proof for the appended term in a substitution snoc
- fullness proof for coherence terms

## Types, Contexts, Terms, and Substitutions

Types are still generated by `⋆` and hom-types `[ A ] t ⇒ u :[ p , q ]`.
Contexts are still lists of typed declarations.

The main CaTT-specific term former is:
- `coh Γps A u v p q pf σ`, representing the coherence determined by a full
  pasting context `Γps`, base type `A`, endpoints `u`,`v`, and substitution `σ`

As in the preCaTT layer, substitutions are not introduced via extra
primitive "normalized" constructors. Instead, we keep a small inductive syntax
(`∅S`, `⟨_,_⟩∶[_]`) and define weakening, substitution, composition, and
identity as *computing recursive functions*.

We keep only a translation to raw pre-syntax (`1a-preCaTT`) for proof
obligations and structural lemmas.

The rest of the file should be read with that split in mind: first the
raw-indexed intrinsic layer, then the public packaging layer built on top of
it.

```agda
module 2a-CaTT-V9 where
```

## Data Declarations and Translation Functions

The syntax extends the preCaTT raw syntax with explicit typing/fullness proofs,
with `coh` as the CaTT-specific constructor.

In this V9 refactor, the raw-indexed proof layer is no longer hidden inside a
local implementation module. Instead, we define it directly at the top level
under the primed names `Ty'`, `Tm'`, and `Sub'`. This keeps the proof-relevant
structure visible from the start and avoids the later re-export step.

```agda

infixl 8  _[_]T' _[_]t'
infixr 10 _∘'_

mutual
  data Ctx-iswf : Pre.Ctx → Set
  data Ty-iswf  : ∀ {Γ} → Pre.Ty Γ → Set
  data Tm-iswf  : ∀ {Γ} → Pre.Tm Γ → Set
  data Sub-iswf : ∀ {Γ Δ} → Pre.Sub Γ Δ → Set

  record Ty' (Γ : Pre.Ctx) : Set where
    pattern
    constructor mkTy
    inductive
    field
      rawTy : Pre.Ty Γ
      ty-iswf : Ty-iswf rawTy 

  record Tm' (Γ : Pre.Ctx) : Set where
    pattern
    constructor mkTm
    inductive
    field
      rawTm : Pre.Tm Γ
      tm-iswf : Tm-iswf rawTm

  record Sub' (Γ Δ : Pre.Ctx) : Set where
    pattern
    constructor mkSub
    inductive
    field
      rawSub : Pre.Sub Γ Δ
      sub-iswf : Sub-iswf rawSub

  Raw-Ty' : ∀ {Γ} → Ty' Γ → Pre.Ty Γ
  Raw-Ty' = Ty'.rawTy

  Ty-wf' : ∀ {Γ} → (A : Ty' Γ) → Ty-iswf (Raw-Ty' A)
  Ty-wf' = Ty'.ty-iswf

  Raw-Tm' : ∀ {Γ} → Tm' Γ → Pre.Tm Γ
  Raw-Tm' = Tm'.rawTm

  Tm-wf' : ∀ {Γ} → (t : Tm' Γ) → Tm-iswf (Raw-Tm' t)
  Tm-wf' = Tm'.tm-iswf

  Raw-Sub' : ∀ {Γ Δ} → Sub' Γ Δ → Pre.Sub Γ Δ
  Raw-Sub' = Sub'.rawSub

  Sub-wf' : ∀ {Γ Δ} → (σ : Sub' Γ Δ) → Sub-iswf (Raw-Sub' σ)
  Sub-wf' = Sub'.sub-iswf

  TmTyEq' : ∀ {Γ} → Tm' Γ → Ty' Γ → Set
  TmTyEq' t A = Pre.tyOf (Raw-Tm' t) ≡ Raw-Ty' A

  TmTyEqSub' : ∀ {Γ Δ} → Tm' Γ → Ty' Δ → Sub' Γ Δ → Set
  TmTyEqSub' {Γ} {Δ} t A σ =
    Pre.tyOf (Raw-Tm' t) ≡
    Pre._[_]T (Raw-Ty' A) (Raw-Sub' σ)

  []T-iswf
    : ∀ {Γ Δ} → (A : Ty' Δ) → (σ : Sub' Γ Δ)
    → Ty-iswf (Pre._[_]T (Raw-Ty' A) (Raw-Sub' σ))
  _[_]T'
    : ∀ {Γ Δ} → Ty' Δ → Sub' Γ Δ → Ty' Γ
  []t-iswf
    : ∀ {Γ Δ} → (t : Tm' Δ) → (σ : Sub' Γ Δ)
    → Tm-iswf (Pre._[_]t (Raw-Tm' t) (Raw-Sub' σ))
  _[_]t'
    : ∀ {Γ Δ} → Tm' Δ → Sub' Γ Δ → Tm' Γ
  _∘'_
    : ∀ {Γ Δ Θ} → Sub' Δ Θ → Sub' Γ Δ → Sub' Γ Θ

  data Ctx-iswf where
    ∅wf : Ctx-iswf Pre.∅
    _,_wf : ∀ {Γ : Pre.Ctx} → Ctx-iswf Γ → (A : Ty' Γ) → Ctx-iswf (Γ Pre., Raw-Ty' A) 

  data Ty-iswf where
    ⋆wf : ∀ {Γ} → Ty-iswf {Γ} Pre.⋆
    hom-wf : ∀ {Γ} → (A : Ty' Γ) → {t u : Tm' Γ}
      → TmTyEq' t A → TmTyEq' u A
      → Ty-iswf (Pre.[ Raw-Ty' A ] (Raw-Tm' t) ⇒ (Raw-Tm' u))

  data Sub-iswf where
    ∅Swf : ∀ {Γ} → Sub-iswf (Pre.∅S {Γ})
    ⟨_,_⟩:[_]wf : ∀ {Γ Δ} {A : Ty' Δ} → (σ : Sub' Γ Δ) → (t : Tm' Γ)
      → TmTyEqSub' {Γ} {Δ} t A σ
      → Sub-iswf (Pre.⟨_,_⟩ {A = Raw-Ty' A} (Raw-Sub' σ) (Raw-Tm' t))

  ∘-iswf-raw : ∀ {Γ Δ Θ} {τ : Pre.Sub Δ Θ}
    → Sub-iswf τ → (σ : Sub' Γ Δ)
    → Sub-iswf (Pre._∘_ τ (Raw-Sub' σ))
  ∘-iswf-raw {τ = Pre.∅S} τwf σ = ∅Swf
  ∘-iswf-raw
    {τ = Pre.⟨ τ , t ⟩}
    (⟨_,_⟩:[_]wf {A = A} (mkSub .τ τwf) (mkTm .t twf) r)
    (mkSub σ σwf) =
      ⟨_,_⟩:[_]wf {A = A}
        (mkSub (τ Pre.∘ σ) (∘-iswf-raw τwf (mkSub σ σwf)))
        (mkTm (t Pre.[ σ ]t) ([]t-iswf-raw twf (mkSub σ σwf)))
        (trans
          (tyOf-comm t ((Raw-Ty' A) Pre.[ τ ]T) (mkSub σ σwf) r)
          (Pre.[∘]T (Raw-Ty' A) τ σ))

  ∘-iswf : ∀ {Γ Δ Θ} → (τ : Sub' Δ Θ) → (σ : Sub' Γ Δ)
    → Sub-iswf (Pre._∘_ (Raw-Sub' τ) (Raw-Sub' σ))
  ∘-iswf (mkSub τ τwf) σ = ∘-iswf-raw τwf σ

  mkSub τ τwf ∘' mkSub σ σwf = mkSub (τ Pre.∘ σ) (∘-iswf-raw τwf (mkSub σ σwf))

  tyOf-lookup-comm : ∀ {Γ Δ} → (x : Pre.Var Δ) → (σ : Sub' Γ Δ)
    → Pre.tyOf (Pre.lookup x (Raw-Sub' σ)) ≡ Pre._[_]T (Pre.var-to-type x) (Raw-Sub' σ)
  tyOf-lookup-comm Pre.vz (mkSub Pre.⟨ σ , t ⟩ (⟨_,_⟩:[_]wf {A = A} (mkSub .σ σwf) (mkTm .t twf) p)) =
    trans p (sym (Pre.wkTy-sub (Raw-Ty' A) σ t))
  tyOf-lookup-comm (Pre.vs x) (mkSub Pre.⟨ σ , t ⟩ ⟨ mkSub _ σwf , mkTm _ _ ⟩:[ _ ]wf) =
    trans (tyOf-lookup-comm x (mkSub σ σwf)) (sym (Pre.wkTy-sub (Pre.var-to-type x) σ t))

  tyOf-comm : ∀ {Γ Δ} → (t : Pre.Tm Δ) → (A : Pre.Ty Δ) → (σ : Sub' Γ Δ)
    → Pre.tyOf t ≡ A → Pre.tyOf (t Pre.[ Raw-Sub' σ ]t) ≡ A Pre.[ Raw-Sub' σ ]T
  tyOf-comm (Pre.var x) A σ p =
    trans (tyOf-lookup-comm x σ) (cong (λ X → X Pre.[ Raw-Sub' σ ]T) p)
  tyOf-comm (Pre.coh-raw B u v τ) A σ p =
    trans (sym (Pre.[∘]T (Pre.[ B ] u ⇒ v) τ (Raw-Sub' σ))) (cong (λ X → X Pre.[ Raw-Sub' σ ]T) p)

  []T-iswf-raw :
    ∀ {Γ Δ} →
    (A : Pre.Ty Δ) →
    Ty-iswf A →
    (σ : Sub' Γ Δ) →
    Ty-iswf (Pre._[_]T A (Raw-Sub' σ))
  []T-iswf-raw Pre.⋆ p σ = ⋆wf
  []T-iswf-raw {Γ} (Pre.[ A ] t ⇒ u) (hom-wf (mkTy _ Awf) {mkTm _ twf} {mkTm _ uwf} p p') (mkSub σ σwf) =
    hom-wf {Γ}
      (mkTy (A Pre.[ σ ]T) ([]T-iswf (mkTy A Awf) (mkSub σ σwf)))
      {mkTm (t Pre.[ σ ]t) ([]t-iswf (mkTm t twf) (mkSub σ σwf))}
      {mkTm (u Pre.[ σ ]t) ([]t-iswf (mkTm u uwf) (mkSub σ σwf))}
      (tyOf-comm t A (mkSub σ σwf) p)
      (tyOf-comm u A (mkSub σ σwf) p')

  []T-iswf (mkTy A Awf) σ = []T-iswf-raw A Awf σ

  A [ σ ]T' = mkTy ((Raw-Ty' A) Pre.[ Raw-Sub' σ ]T) ([]T-iswf A σ) 

  data Tm-iswf where
    var-wf : ∀ {Γ} → (x : Pre.Var Γ) → Tm-iswf (Pre.var x)
    coh-wf : ∀ {Γ Δ} → (ps : CtxPs Δ) → (A : Ty' Δ) → (u v : Tm' Δ)
      → (Pre.tyOf (Raw-Tm' u) ≡ (Raw-Ty' A)) → (Pre.tyOf (Raw-Tm' v) ≡ (Raw-Ty' A))
      → (is-full ps (Raw-Tm' u) (Raw-Tm' v) ≡ true) → (σ : Sub' Γ Δ)
      → Tm-iswf (Pre.coh-raw (Raw-Ty' A) (Raw-Tm' u) (Raw-Tm' v) (Raw-Sub' σ))

  lookup-iswf : ∀ {Γ Δ} → (x : Pre.Var Δ) → (σ : Sub' Γ Δ)
    → Tm-iswf (Pre.lookup x (Raw-Sub' σ))
  lookup-iswf Pre.vz (mkSub Pre.⟨ _ , _ ⟩ ⟨ mkSub _ _  , mkTm _ twf ⟩:[ _ ]wf) = twf
  lookup-iswf (Pre.vs x) (mkSub Pre.⟨ σ , _ ⟩ ⟨ mkSub _ σwf , mkTm _ _ ⟩:[ p ]wf) =
    lookup-iswf x (mkSub σ σwf)  

  []t-iswf-raw :
    ∀ {Γ Δ} {t : Pre.Tm Δ} →
    Tm-iswf t →
    (σ : Sub' Γ Δ) →
    Tm-iswf (Pre._[_]t t (Raw-Sub' σ))
  []t-iswf-raw (var-wf x) σ = lookup-iswf x σ
  []t-iswf-raw
    (coh-wf ps (mkTy A Awf) (mkTm t twf) (mkTm u uwf) p q r (mkSub γ γwf))
    (mkSub σ σwf) =
      coh-wf ps (mkTy A Awf) (mkTm t twf) (mkTm u uwf) p q r
        (mkSub (γ Pre.∘ σ) (∘-iswf-raw γwf (mkSub σ σwf)))

  []t-iswf (mkTm t twf) σ = []t-iswf-raw twf σ

  t [ σ ]t' = mkTm ((Raw-Tm' t) Pre.[ Raw-Sub' σ ]t) ([]t-iswf t σ)

```

The primed records and projections are already the public raw-indexed layer, so
no extra re-export step is needed.

```agda
open Ty' public
open Tm' public
open Sub' public
```

## Public Context Layer

The public type `Ctx` packages a raw context together with a proof that it is
well formed. Public types, terms, and substitutions are then the same kind of
proof-carrying records as before, but now indexed by `Ctx` instead of bare
`Pre.Ctx`.

The `cast-*` maps move from the public layer back to the raw-indexed primed
layer. They are the basic bridge used throughout the rest of the file: the raw
proof layer is now exposed directly, but many public constructions are still
most convenient to phrase through these casts.

```agda

record Ctx : Set where
  pattern
  constructor mkCtx
  inductive
  field
    rawCtx : Pre.Ctx
    ctx-iswf : Ctx-iswf rawCtx

Raw-Ctx : Ctx → Pre.Ctx
Raw-Ctx = Ctx.rawCtx

Ctx-wf : (Γ : Ctx) → Ctx-iswf (Raw-Ctx Γ)
Ctx-wf = Ctx.ctx-iswf

record Ty (Γ : Ctx) : Set where
  pattern
  constructor mkTy
  field
    rawTy   : Pre.Ty (Raw-Ctx Γ)
    ty-iswf : Ty-iswf rawTy

record Tm (Γ : Ctx) : Set where
  pattern
  constructor mkTm
  field
    rawTm   : Pre.Tm (Raw-Ctx Γ)
    tm-iswf : Tm-iswf rawTm

record Sub (Γ Δ : Ctx) : Set where
  pattern
  constructor mkSub
  field
    rawSub   : Pre.Sub (Raw-Ctx Γ) (Raw-Ctx Δ)
    sub-iswf : Sub-iswf rawSub

Raw-Ty : ∀ {Γ} → Ty Γ → Pre.Ty (Raw-Ctx Γ)
Raw-Ty = Ty.rawTy

Ty-wf : ∀ {Γ} → (A : Ty Γ) → Ty-iswf (Raw-Ty A)
Ty-wf = Ty.ty-iswf

Raw-Tm : ∀ {Γ} → Tm Γ → Pre.Tm (Raw-Ctx Γ)
Raw-Tm = Tm.rawTm

Tm-wf : ∀ {Γ} → (t : Tm Γ) → Tm-iswf (Raw-Tm t)
Tm-wf = Tm.tm-iswf

Raw-Sub : ∀ {Γ Δ} → Sub Γ Δ → Pre.Sub (Raw-Ctx Γ) (Raw-Ctx Δ)
Raw-Sub = Sub.rawSub

Sub-wf : ∀ {Γ Δ} → (σ : Sub Γ Δ) → Sub-iswf (Raw-Sub σ)
Sub-wf = Sub.sub-iswf

Var : Ctx → Set
Var Γ = Pre.Var (Raw-Ctx Γ)

∅ : Ctx
∅ = mkCtx Pre.∅ ∅wf

cast-ty : {Γ : Ctx} → Ty Γ → Ty' (Raw-Ctx Γ)
cast-ty A = mkTy (Raw-Ty A) (Ty-wf A)

cast-tm : {Γ : Ctx} → Tm Γ → Tm' (Raw-Ctx Γ)
cast-tm t = mkTm (Raw-Tm t) (Tm-wf t)

cast-sub : {Γ Δ : Ctx} → Sub Γ Δ → Sub' (Raw-Ctx Γ) (Raw-Ctx Δ)
cast-sub σ = mkSub (Raw-Sub σ) (Sub-wf σ)

_,_ : (Γ : Ctx) → Ty Γ → Ctx
Γ , A = mkCtx (Raw-Ctx Γ Pre., Raw-Ty A) (_,_wf (Ctx-wf Γ) (cast-ty A))

⋆ : ∀ {Γ} → Ty Γ
⋆ = mkTy Pre.⋆ ⋆wf

∅S : ∀ {Γ} → Sub Γ ∅
∅S = mkSub Pre.∅S ∅Swf

TmTyEq : ∀ {Γ} → Tm Γ → Ty Γ → Set
TmTyEq t A = Pre.tyOf (Raw-Tm t) ≡ Raw-Ty A

TmTyEqSub : ∀ {Γ Δ} → Tm Γ → Ty Δ → Sub Γ Δ → Set
TmTyEqSub {Γ} {Δ} t A σ =
  Pre.tyOf (Raw-Tm t) ≡
  Pre._[_]T {Γ = Raw-Ctx Γ} {Δ = Raw-Ctx Δ} (Raw-Ty A) (Raw-Sub σ)

data CtxView : Ctx → Set where
  ∅-view : CtxView ∅
  snoc-view : ∀ {Γ} (A : Ty Γ) → CtxView (Γ , A)

ctx-view : (Γ : Ctx) → CtxView Γ
ctx-view (mkCtx Pre.∅ ∅wf) = ∅-view
ctx-view (mkCtx _ (_,_wf {Γ = Γraw} Γwf A0)) =
  snoc-view {Γ = mkCtx Γraw Γwf} (mkTy (Raw-Ty' A0) (Ty-wf' A0))

```

## Public Weakening and Type Formation

The next block reconstructs weakening and `tyOf` on the public layer. Each
definition has a proof-producing part, such as `wkTy-iswf` or `tyOf-iswf`, and
then the public operation itself, such as `wkTy`, `wkTm`, `wkSub`, or `tyOf`.

As in `1a-preCaTT`, the computational content still comes from the raw
pre-syntax. The extra work here is exactly the proof plumbing showing that each
raw operation preserves well-formedness.

```agda
mutual
  wkTy : ∀ {Γ} {A : Ty Γ} → Ty Γ → Ty (Γ , A)
  wkTm : ∀ {Γ} {A : Ty Γ} → Tm Γ → Tm (Γ , A)
  wkSub : ∀ {Γ Δ} {A : Ty Γ} → Sub Γ Δ → Sub (Γ , A) Δ
  
  wkTy-iswf : ∀ {Γ : Ctx} {A : Ty Γ} (B : Ty Γ)
    → Ty-iswf (Pre.wkTy {A = Raw-Ty A} (Raw-Ty B))

  wkTyCommRaw : ∀ {Γ : Ctx} {B : Ty Γ} → (A : Ty Γ) →
    Raw-Ty {Γ , B} (wkTy {A = B} A) ≡ Pre.wkTy (Raw-Ty A)

  var-to-type-iswf' : (Γ : Ctx) → (x : Pre.Var (Raw-Ctx Γ))
    → Ty-iswf (Pre.var-to-type x)

  var-to-type-iswf : {Γ : Ctx} → (x : Var Γ)
    → Ty-iswf (Pre.var-to-type x)

  tyOf-iswf : {Γ : Ctx} → (t : Tm Γ) → Ty-iswf (Pre.tyOf (Raw-Tm t))
  tyOf : ∀ {Γ} → Tm Γ → Ty Γ

  wkTy-iswf-raw
    : ∀ {Γ : Ctx} {A : Ty Γ} (B : Pre.Ty (Raw-Ctx Γ))
    → Ty-iswf B → Ty-iswf (Pre.wkTy {A = Raw-Ty A} B)
  wkTm-iswf
    : ∀ {Γ : Ctx} {A : Ty Γ} (t : Tm Γ)
    → Tm-iswf (Pre.wkTm {A = Raw-Ty A} (Raw-Tm t))
  wkSub-iswf-raw
    : ∀ {Γ : Ctx} {Δ : Pre.Ctx} {A : Ty Γ}
      (σ : Pre.Sub (Raw-Ctx Γ) Δ)
    → Sub-iswf σ
    → Sub-iswf (Pre.wkSub {A = Raw-Ty A} σ)
  wkSub-iswf
    : ∀ {Γ Δ : Ctx} {A : Ty Γ} (σ : Sub Γ Δ)
    → Sub-iswf (Pre.wkSub {A = Raw-Ty A} (Raw-Sub σ))

  wkTy-iswf-raw Pre.⋆ ⋆wf = ⋆wf
  wkTy-iswf-raw {Γ} {A} (Pre.[ B ] t ⇒ u) (hom-wf (mkTy .B Bwf) {mkTm .t twf} {mkTm .u uwf} q r) =
    hom-wf {Γ = Raw-Ctx (Γ , A)}
      (mkTy (Pre.wkTy {A = Raw-Ty A} B) (wkTy-iswf-raw {Γ = Γ} {A = A} B Bwf))
      {mkTm (Pre.wkTm {A = Raw-Ty A} t) (wkTm-iswf {Γ = Γ} {A = A} (mkTm t twf))}
      {mkTm (Pre.wkTm {A = Raw-Ty A} u) (wkTm-iswf {Γ = Γ} {A = A} (mkTm u uwf))}
      (trans (Pre.tyOf-wk t) (cong Pre.wkTy q))
      (trans (Pre.tyOf-wk u) (cong Pre.wkTy r))

  wkTm-iswf (mkTm (Pre.var x) (var-wf .x)) = var-wf (Pre.vs x)
  wkTm-iswf {Γ} {A = A'} (mkTm
    (Pre.coh-raw B u v σRaw)
    (coh-wf {Δ = Δ} ps (mkTy .B Bwf) (mkTm .u uwf) (mkTm .v vwf) p q r (mkSub .σRaw σwf))) =
    coh-wf {Γ = Raw-Ctx (Γ , A')} {Δ = Δ} ps (mkTy B Bwf) (mkTm u uwf) (mkTm v vwf) p q r
      (mkSub (Pre.wkSub σRaw) (wkSub-iswf-raw {Γ = Γ} {A = A'} σRaw σwf))

  wkTy-iswf {A = A} B = wkTy-iswf-raw {A = A} (Raw-Ty B) (Ty-wf B)

  wkTy {A = A} B = mkTy (Pre.wkTy (Raw-Ty B)) (wkTy-iswf {A = A} B)

  wkTyCommRaw A = refl

  var-to-type-iswf' (mkCtx Pre.∅ ∅wf) ()
  var-to-type-iswf' (mkCtx _ (_,_wf {Γ = Γraw} Γwf A0)) Pre.vz =
    wkTy-iswf
      {Γ = mkCtx Γraw Γwf}
      {A = mkTy (Raw-Ty' A0) (Ty-wf' A0)}
      (mkTy (Raw-Ty' A0) (Ty-wf' A0))
  var-to-type-iswf' (mkCtx _ (_,_wf {Γ = Γraw} Γwf A0)) (Pre.vs x) =
    wkTy-iswf
      {Γ = mkCtx Γraw Γwf}
      {A = mkTy (Raw-Ty' A0) (Ty-wf' A0)}
      (mkTy (Pre.var-to-type x) (var-to-type-iswf' (mkCtx Γraw Γwf) x))

  var-to-type-iswf {Γ} x = var-to-type-iswf' Γ x

  tyOf-iswf {Γ} (mkTm (Pre.var x) twf) = var-to-type-iswf {Γ} x
  tyOf-iswf (mkTm (Pre.coh-raw A t u σ)
    (coh-wf _ (mkTy .A Awf) (mkTm .t twf) (mkTm .u uwf) p q r (mkSub .σ σwf))) =
      []T-iswf
        (mkTy (Pre.[ A ] t ⇒ u) (hom-wf (mkTy A Awf) {mkTm t twf} {mkTm u uwf} p q))
        (mkSub σ σwf)

  tyOf t = mkTy (Pre.tyOf (Raw-Tm t)) (tyOf-iswf t)

  wkTm {Γ} {A} (mkTm (Pre.var x) p) =
    mkTm (Pre.var (Pre.vs x)) (var-wf (Pre.vs x))
  wkTm {Γ} {A} (mkTm (Pre.coh-raw B u v σRaw)
    (coh-wf {Δ = Δ} ps (mkTy .B Bwf) (mkTm .u uwf) (mkTm .v vwf) p q pf (mkSub .σRaw σwf))) =
    mkTm (Pre.coh-raw B u v (Pre.wkSub σRaw))
      (coh-wf {Γ = Raw-Ctx (Γ , A)} {Δ = Δ} ps (mkTy B Bwf) (mkTm u uwf) (mkTm v vwf) p q pf
        (mkSub (Pre.wkSub σRaw) (wkSub-iswf-raw {Γ = Γ} {A = A} σRaw σwf)))

  wkSub {Γ} {Δ} {A} σ =
    mkSub (Pre.wkSub {A = Raw-Ty A} (Raw-Sub σ)) (wkSub-iswf {Γ} {Δ} {A} σ)

  wkSub-iswf-raw Pre.∅S ∅Swf = ∅Swf
  wkSub-iswf-raw {Γ} {A = A}
    (Pre.⟨ σRaw , tRaw ⟩) q with q
  ... | (⟨_,_⟩:[_]wf {A = B} (mkSub .σRaw σwf) (mkTm .tRaw twf) r) =
    ⟨_,_⟩:[_]wf {A = B}
      (mkSub (Pre.wkSub σRaw) (wkSub-iswf-raw {Γ} {A = A} σRaw σwf))
      (mkTm (Pre.wkTm tRaw) (wkTm-iswf {Γ} {A} (mkTm tRaw twf)))
      (trans
        (Pre.tyOf-wk tRaw)
        (trans
          (cong Pre.wkTy r)
          (sym (Pre.wkTy-[]T (Raw-Ty' B) σRaw))))

  wkSub-iswf {Γ} {Δ} {A} (mkSub σ σwf) = wkSub-iswf-raw {Γ} {Raw-Ctx Δ} {A} σ σwf

_[_]T : ∀ {Γ Δ} → Ty Δ → Sub Γ Δ → Ty Γ
A [ σ ]T = mkTy ((Raw-Ty A) Pre.[ Raw-Sub σ ]T) ([]T-iswf (cast-ty A) (cast-sub σ))

Raw-Ty-Sub : ∀ {Γ Δ} → (A : Ty Δ) → (σ : Sub Γ Δ)
  → Raw-Ty (A [ σ ]T) ≡ Pre._[_]T (Raw-Ty A) (Raw-Sub σ)
Raw-Ty-Sub A σ = refl

_[_]t : ∀ {Γ Δ} → Tm Δ → Sub Γ Δ → Tm Γ
t [ σ ]t = mkTm ((Raw-Tm t) Pre.[ Raw-Sub σ ]t) ([]t-iswf (cast-tm t) (cast-sub σ))

_∘s_ : ∀ {Γ Δ Θ} → Sub Δ Θ → Sub Γ Δ → Sub Γ Θ
τ ∘s σ = mkSub ((Raw-Sub τ) Pre.∘ (Raw-Sub σ)) (∘-iswf (cast-sub τ) (cast-sub σ))

_∘_ : ∀ {Γ Δ Θ} → Sub Δ Θ → Sub Γ Δ → Sub Γ Θ
_∘_ = _∘s_

Raw-Sub-assoc :
  ∀ {Γ Δ Θ Ξ}
  → (γ : Sub Θ Ξ) (τ : Sub Δ Θ) (σ : Sub Γ Δ)
  → Raw-Sub ((γ ∘ τ) ∘ σ) ≡ Raw-Sub (γ ∘ (τ ∘ σ))
Raw-Sub-assoc γ τ σ = Pre.assocS (Raw-Sub γ) (Raw-Sub τ) (Raw-Sub σ)

[_]_⇒_:[_,_] : {Γ : Ctx} → (A : Ty Γ) → (t u : Tm Γ) → (tyOf {Γ} t ≡ A) → (tyOf {Γ} u ≡ A) → Ty Γ
[_]_⇒_:[_,_] {Γ} A t u p q =
  mkTy (Pre.[ Raw-Ty A ] (Raw-Tm t) ⇒ (Raw-Tm u))
    (hom-wf {Γ = Raw-Ctx Γ} (cast-ty A) {cast-tm t} {cast-tm u} (cong Raw-Ty p) (cong Raw-Ty q))

⟨_,_⟩∶[_] : ∀ {Γ Δ} {A : Ty Δ} → (σ : Sub Γ Δ) → (t : Tm Γ)
  → (Pre.tyOf (Raw-Tm t) ≡ Raw-Ty A Pre.[ Raw-Sub σ ]T) → Sub Γ (Δ , A)
⟨_,_⟩∶[_] {Γ} {Δ} {A} σ t p =
  mkSub (Pre.⟨ Raw-Sub σ , Raw-Tm t ⟩)
    (⟨_,_⟩:[_]wf {A = cast-ty A} (cast-sub σ) (cast-tm t) p)

coh : ∀ {Γ Δ} → (ps : CtxPs (Raw-Ctx Δ)) → (A : Ty Δ) → (u v : Tm Δ)
  → (Pre.tyOf (Raw-Tm u) ≡ (Raw-Ty A)) → (Pre.tyOf (Raw-Tm v) ≡ (Raw-Ty A))
  → (is-full ps (Raw-Tm u) (Raw-Tm v) ≡ true) → (σ : Sub Γ Δ)
  → Tm Γ
coh {Γ} {Δ} ps A u v p q r σ =
  mkTm (Pre.coh-raw (Raw-Ty A) (Raw-Tm u) (Raw-Tm v) (Raw-Sub σ))
    (coh-wf {Γ = Raw-Ctx Γ} {Δ = Raw-Ctx Δ} ps (cast-ty A) (cast-tm u) (cast-tm v) p q r (cast-sub σ))

zero : ∀ {Γ A} → Var (Γ , A)
zero = Pre.vz

succ : ∀ {Γ A} → Var Γ → Var (Γ , A)
succ = Pre.vs

vz : {Γ : Ctx} → {A : Ty Γ} → Tm (Γ , A)
vz = mkTm (Pre.var Pre.vz) (var-wf Pre.vz)

vs : {Γ : Ctx} → {A : Ty Γ} → (x : Var Γ) → Tm (Γ , A)
vs x = mkTm (Pre.var (Pre.vs x)) (var-wf (Pre.vs x))

var : ∀ {Γ} → Var Γ → Tm Γ
var x = mkTm (Pre.var x) (var-wf x)

lookup : ∀ {Γ Δ} → Var Δ → Sub Γ Δ → Tm Γ
lookup x σ = mkTm (Pre.lookup x (Raw-Sub σ)) (lookup-iswf x (cast-sub σ))
```

## Identity and Canonical Weakening Maps

The last block defines identity substitution and the canonical weakening maps
`wk` and `wk³`. As in `1a-preCaTT`, these are computed operations rather than
primitive constructors. That matters later: many equalities involving identity
and weakening reduce by computation once the accompanying well-formedness proof
has been built.

```agda
mutual
  idS : (Γ : Ctx) → Sub Γ Γ
  idS-iswf : (Γ : Ctx) → Sub-iswf (Pre.idS (Raw-Ctx Γ))

  idS Γ = mkSub (Pre.idS (Raw-Ctx Γ)) (idS-iswf Γ)

  idS-iswf (mkCtx Pre.∅ ∅wf) = ∅Swf
  idS-iswf (mkCtx _ (_,_wf {Γ = Γraw} Γwf A0)) =
    let
      Γ' : Ctx
      Γ' = mkCtx Γraw Γwf

      A' : Ty Γ'
      A' = mkTy (Raw-Ty' A0) (Ty-wf' A0)
    in
      ⟨_,_⟩:[_]wf {A = cast-ty A'}
        (cast-sub (wkSub {Γ = Γ'} {Δ = Γ'} {A = A'} (idS Γ')))
        (cast-tm (var {Γ = Γ' , A'} Pre.vz))
        (sym
          (trans
            (Pre.wkTy-[]T (Raw-Ty A') (Raw-Sub (idS Γ')))
            (cong Pre.wkTy (Pre.[idS]T (Raw-Ty A')))))

wk : ∀ {Γ} {A : Ty Γ} → Sub (Γ , A) Γ
wk {Γ} {A} = wkSub {Γ = Γ} {Δ = Γ} {A = A} (idS Γ)

wk³ :
  ∀ {Γ}
  {A : Ty Γ}
  {B : Ty (Γ , A)}
  {C : Ty ((Γ , A) , B)}
  → Sub (((Γ , A) , B) , C) Γ
wk³ {Γ} {A} {B} {C} =
  _∘s_
    {Γ = (((Γ , A) , B) , C)}
    {Δ = (Γ , A)}
    {Θ = Γ}
    (wk {Γ = Γ} {A = A})
    (_∘s_
      {Γ = (((Γ , A) , B) , C)}
      {Δ = ((Γ , A) , B)}
      {Θ = (Γ , A)}
      (wk {Γ = Γ , A} {A = B})
      (wk {Γ = (Γ , A) , B} {A = C}))

var-eq : ∀ {Γ} → Var Γ → Var Γ → Bool
var-eq x y = Pre.var-eq x y

∃-var : (Γ : Ctx) → (Var Γ → Bool) → Bool
∃-var Γ P = Pre.∃-var (Raw-Ctx Γ) P

∀-var : (Γ : Ctx) → (Var Γ → Bool) → Bool
∀-var Γ P = Pre.∀-var (Raw-Ctx Γ) P

var-to-type : ∀ {Γ} → Var Γ → Ty Γ
var-to-type {Γ} x = mkTy (Pre.var-to-type x) (var-to-type-iswf {Γ} x)

abstract
  mutual
    Ctx-iswf-uip : (Γ : Pre.Ctx) → (p q : Ctx-iswf Γ) → p ≡ q
    Ty-iswf-uip : {Γ : Pre.Ctx} → (A : Pre.Ty Γ) → (p q : Ty-iswf A) → p ≡ q
    Tm-iswf-uip : {Γ : Pre.Ctx} → (t : Pre.Tm Γ) → (p q : Tm-iswf t) → p ≡ q
    Sub-iswf-uip : {Γ Δ : Pre.Ctx} → (σ : Pre.Sub Γ Δ) → (p q : Sub-iswf σ) → p ≡ q

    uip : ∀ {ℓ} {A : Set ℓ} {x y : A} (p q : x ≡ y) → p ≡ q
    uip refl refl = refl

    Ctx-iswf-uip .Pre.∅ ∅wf ∅wf = refl
    Ctx-iswf-uip
      (Γ Pre., Araw)
      (_,_wf {Γ = Γ} Γwf (mkTy .Araw Awf))
      (_,_wf {Γ = .Γ} Γwf' (mkTy .Araw Awf'))
      rewrite
        Ctx-iswf-uip Γ Γwf Γwf'
      | Ty-iswf-uip Araw Awf Awf'
      = refl

    Ty-iswf-uip Pre.⋆ ⋆wf ⋆wf = refl
    Ty-iswf-uip
      (Pre.[ A ] t ⇒ u)
      (hom-wf {Γ = Γ} (mkTy .A Awf) {mkTm .t twf} {mkTm .u uwf} q r)
      (hom-wf {Γ = .Γ} (mkTy .A Awf') {mkTm .t twf'} {mkTm .u uwf'} q' r')
      rewrite
        Ty-iswf-uip A Awf Awf'
      | Tm-iswf-uip t twf twf'
      | Tm-iswf-uip u uwf uwf'
      | uip q q'
      | uip r r' = refl

    postulate
      CtxPs-uip : {Γ : Pre.Ctx} → (p q : PrePs.CtxPs Γ) → p ≡ q

    Tm-iswf-uip (Pre.var x) (var-wf .x) (var-wf .x) = refl
    Tm-iswf-uip
      (Pre.coh-raw A t u σ)
      (coh-wf ps (mkTy .A Awf) (mkTm .t twf) (mkTm .u uwf) p q r (mkSub .σ σwf))
      (coh-wf ps' (mkTy .A Awf') (mkTm .t twf') (mkTm .u uwf') p' q' r' (mkSub .σ σwf'))
      rewrite
        Ty-iswf-uip A Awf Awf'
      | Tm-iswf-uip t twf twf'
      | Tm-iswf-uip u uwf uwf'
      | uip p p'
      | uip q q'
      | CtxPs-uip ps ps'
      | uip r r'
      | Sub-iswf-uip σ σwf σwf'
      = refl

    Sub-iswf-uip Pre.∅S ∅Swf ∅Swf = refl
    Sub-iswf-uip
      (Pre.⟨ σ , t ⟩)
      (⟨_,_⟩:[_]wf {A = mkTy A Awf} (mkSub .σ σwf) (mkTm .t twf) r)
      (⟨_,_⟩:[_]wf {A = mkTy .A Awf'} (mkSub .σ σwf') (mkTm .t twf') r')
      rewrite
        Ty-iswf-uip A Awf Awf'
      | Sub-iswf-uip σ σwf σwf'
      | Tm-iswf-uip t twf twf'
      | uip r r'
      = refl

  mutual
    Ty-ext : {Γ : Ctx} → {A A' : Ty Γ} → (Raw-Ty A ≡ Raw-Ty A') → A ≡ A'
    Tm-ext : {Γ : Ctx} → {t t' : Tm Γ} → (Raw-Tm t ≡ Raw-Tm t') → t ≡ t'
    Sub-ext : {Γ Δ : Ctx} → {σ σ' : Sub Γ Δ} → (Raw-Sub σ ≡ Raw-Sub σ') → σ ≡ σ'

    Ty-ext {Γ = mkCtx Γraw Γwf} {A = mkTy A Awf} {mkTy .A Awf'} refl
      rewrite Ty-iswf-uip A Awf Awf' = refl
    Tm-ext {Γ = mkCtx Γraw Γwf} {t = mkTm t twf} {mkTm .t twf'} refl
      rewrite Tm-iswf-uip t twf twf' = refl
    Sub-ext {Γ = mkCtx Γraw Γwf} {Δ = mkCtx Δraw Δwf}
      {σ = mkSub σ σwf} {mkSub .σ σwf'} refl
      rewrite Sub-iswf-uip σ σwf σwf' = refl

TmTyEq→tyOf≡ :
  ∀ {Γ} {t : Tm Γ} {A : Ty Γ}
  → TmTyEq t A
  → tyOf t ≡ A
TmTyEq→tyOf≡ p = Ty-ext p

TmTyEqSub→tyOf≡ :

  → tyOf t ≡ A [ σ ]T
```
